def <=>(other)
  return 1 unless other.is_a?(self.class)

  a1, a2 = attr, other.attr
  return form <=> other.form if a1 == a2

  a1.empty? ? 1 : a2.empty? ? -1 : begin
    i1, i2 = LA_SORTORDER.values_at(a1, a2)
    i1 ? i2 ? i1 <=> i2 : -1 : i2 ? 1 : a1 <=> a2
  end
end
