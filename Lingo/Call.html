def talk(input, raw = false)
  config.stdin.reopen(
    input.respond_to?(:read) ? input.read : input)

  start

  res = CHANNELS.flat_map { |key|
    io = config.send(key)
    io.rewind

    lines = io.readlines

    io.truncate(0)
    io.rewind

    lines
  }

  return res.join if raw

  res.each { |i| i.chomp! }

  block_given? ? res.map! { |i| yield i } : begin
    res.sort! unless ENV['LINGO_NO_SORT']
    res.uniq!
    res
  end
end
