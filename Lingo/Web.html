def doit
  @r = LINGO[@l].talk(@q) { |_| _ } if @q && !@q.empty?

  case params[:f]
    when 'json'
      to_json(@q, @r)
    when 'text'
      @r &&  @r.join("\n")
    else
      @r &&= @r.join("\n")
      erb :index
  end
end
