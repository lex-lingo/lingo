def process(obj)
  if obj.is_a?(WordForm)
    str = obj.form

    if obj.respond_to?(:lexicals)
      lex = obj.lexicals.first  # TODO

      att = @map[lex.attr] if lex
      str = @format % [str, lex.form, att] if att
    end
  else
    str = obj.to_s
  end

  @lir ? @lir_rec_buf << str : @file.print(str)
end
