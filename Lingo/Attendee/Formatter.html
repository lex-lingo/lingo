def process(obj)
  if obj.is_a?(WordForm)
    str = obj.form

    if obj.is_a?(Word)
      # TODO: which lexical to select? (currently: first)
      obj.each_lex { |lex|
        att = @map[lex.attr] and str = @format % [str, lex.form, att] }
    end
  else
    str = obj.to_s
  end

  @lir ? @lir_rec_buf << str : @file.print(str)
end
